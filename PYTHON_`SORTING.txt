 ##Seperate numbers and alphabets from the string
 
 s=input("Enter the string")
 s1=s2=s3=output=''
 
 for x  in s:
   if x.isalpha(): #To check if character is alphabet or not
     s1=s1+x
   elif x.isdigit():
     s3=s3+x
   else:
     s2=s2+x #All characters except alphabets
 print("s1 --> ",s1)
 print("s2 --> ",s2)
 print("s3 --> ",s3)
 
 for x in sorted(s1): #It sorts the list
   output=output+x
