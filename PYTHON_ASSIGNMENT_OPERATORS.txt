#Compound Assignment Operator

x=100
x+=10 #x = x+5 --> x=5+5 --> 10
print(x)

print(++x)#It is not increment operator, just a sign.

#In PYTHON, increment and decrement operators are not available.
 
print(--x)#treated as sign and not the decrement operator
print(---x)

a=4
a&=5 # It is  a=a&5 --> &=Bitwise operator, =Assignment Operator
print(a)
