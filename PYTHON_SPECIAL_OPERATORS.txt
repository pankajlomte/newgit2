##Special Operators

#1.Identity Operators

a,b=10,10
print(a is b)#"is"operator-> Checks if both references are pointing same Object
print(a is not b)#"ís not" operator
print(id(a))
print(id(b))

print('')

list1=[10,20,30]
list2=[10,20,30]
print(id(list1))#Different address as List is mutable and new object is 
print(id(list2))#created everytime.
print(list1 is list2)#False

#For content comparison
print(list1==list2)#True
print('')

#2.Membership operators
list1 = [10,20,30]
print(10 in list1)#True --> checks if 10 is member of list1
print(10 not in list1)#False
print(70 in list1)#False
