# {} --> replacement operator

name='Pankaj'
salary=20000
gf='Sunny'

print("Hello {0} your salary is {1} and your gf {2} is waiting".format(name,salary,gf))
#Hello Pankaj your salary is 20000 and your gf Sunny is waiting

print("Hello {} your salary is {} and your gf {} is waiting".format(name,salary,gf))#Wihout Indexes

print("Hello {x} your salary is {y} and your gf {z} is waiting".format(z=gf,y=salary,x=name))#
